## 특정 좌표에서의 RGB값 구하기 

>그림을 클릭했을 때 그 클릭 좌표에 해당되는 색상의 RGB값을 구할  수 있다. RGB값을 이용해 네모 박스 안에 클릭 좌표 색상과 동일한 색상이 나타나도록 작업해보았다.

* #### html + css + javascript

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>RGB</title>
    <style>
        canvas{
            float:left;
        }
        #rgb{
         font-weight:bold;
         float:left;
         margin-left:85px;
         position:absolute;
         top:10px;
        }
        #colorbox{
            width:100px;
            height:420px;
            border:1px solid black;
            float:left;
            margin-left:100px;
            position:relative;
            top:30px;
        }
        
    </style>
    <script>
    window.onload = function(){

        var canvas = document.getElementById('mycanvas');
        var ctx = canvas.getContext('2d');

        var img = new Image();
    
        img.onload = function(){
        ctx.drawImage(img,0,0);
        var imageData = ctx.getImageData(0,0,canvas.width,canvas.height);
        imageData.getRGBA = function(m,n,i){
            return this.data[this.width*4*n+4*m+i];
        };
        var rgb = document.getElementById('rgb');
        canvas.onclick = function(event){

            var colorbox = document.getElementById('colorbox');
            var x = event.offsetX;
            var y = event.offsetY;
            var R = to3digit(imageData.getRGBA(x,y,0));
            var G = to3digit(imageData.getRGBA(x,y,1));
            var B = to3digit(imageData.getRGBA(x,y,2));

            rgb.innerHTML = "R:" +R +'\n'
                            +"G:" +G+'\n'
                            +"B:" +B;

           colorbox.style.backgroundColor = 'rgb('+R+','+G+','+B+')';
        };
    };
    img.src ="피카소.jpg";
};

function to3digit(n){
    return("000"+n).slice(-3);
}
    </script>
  

</head>
<body>
    <canvas id="mycanvas" width="400px" height="450px"></canvas>
    <div><output id="rgb"></output></div>
    <div id="colorbox"></div>
    
</body>
</html>
```
